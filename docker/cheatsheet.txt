# list images
docker images

# remove image
docker rmi yass

# create image
docker build -t yass .

# list running containers
docker ps

# list all containers
docker ps -a

# remove container
docker rm yass

# create container
docker create -p 9090:9090 --name yass yass

# start container
docker start yass

# stop container
docker stop yass

# show logs
docker logs -f yass

# list machines (windows)
docker-machine ls
